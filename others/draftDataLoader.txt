def getFile(scene, light):
    direction = light[0]
    color = light[1]
    return os.path.join(dataPath, direction, color, scene)

class ImagesDatasetType1(Dataset):
    def __init__(self, inLight, outLight):
        self.imagesIn = []
        self.imagesOut = []
        for scene in listScenes :
            with open(getFile(scene, inLight), 'r') as fileIn:
                self.imagesIn.append(imread(fileIn))
            with open(getFile(scene, outLight), 'r') as fileOut:
                self.imagesOut.append(imread(fileOut))
    def __len__(self):
        return len(self.imagesIn)
    def __getitem__(self, idx):
        return {'input': self.imagesIn[idx], 'otherIllumination': self.imagesOut[idx]}
